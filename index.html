var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2 - выводит 2, потому что к переменной "а" он прибавляет 1 (++), "а" объявлена 1
d = b++; alert(d); // 1 - выводит 1, потому что b = 1, прибавление 1 (++) происходит после вывода
c = (2 + ++a); alert(c); // 5 - выходит 5, потому что в 1й строчке к "а" прибавилась 1 и "а" стала равным 2, в этой строчке к "а" прибавляется еще 1 (++) и "а" стала равно 3, и в потом к 3 прибавляется 2
d = (2 + b++); alert(d); // 4 - выходит 4, потому что во 2й строчке к "b" прибавилась 1 и "b" стала равным 2, в этой строчке к "b" прибавляется 2 и затем после выводы алерта прибавляется еще 1 которую мы "не увидели"
alert(a); // 3 - выходит 3, потому что в 1й и 3й строчках к "а" прибавилось 1 и 1 соответственно, а изначально "а" была равна 1, итог а = 1 + 1 + 1 = 3
alert(b); // 3 - выходит 3, потому что в 2й и 4й строчках к "b" прибавилось 1 и 1 соответственно, а изначально "b" была равна 1, итог а = 1 + 1 + 1 = 3

5. *Самостоятельно разобраться с атрибутами тега script (async и defer).

Ответ: Атрибуты применяются только ко внешним (подключаемым) скриптам. Оба атрибута позволяют подгружать содержимое страницы и запустятся только когда подгрузятся, но у атрибута defer есть отличия от async.
Первое - это гарантия, что относительный порядок скриптов с defer будет сохранён.
Второе - атрибут "ждет" когда вся страница загрузится, а async нет
